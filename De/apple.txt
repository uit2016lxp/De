#include<stdio.h>
int main()
{
	int n,i,j,k=0,h,a[17],y;
	scanf("%d",&n);
	for(i=0;i<17;i++)
	{
		for(j=0;j<14;j++)
		{
			if(n==6*i+8*j)
				for(h=0;h<17;h++)
				{
					a[h]=i+j;
					k++;
				}
		}
	}
	if(k==0)
		printf("-1\n");
	else
	{
		for(i=0;i<k-1;i++)
		{
			for(j=0;j<k-1-i;j++)
			{
				if(a[j]>a[i+1])
				{
					y=a[j];
					a[j]=a[j+1];
					a[j+1]=y;
				}
			}
		}
		printf("%d\n",a[0]);
	}
	return 0;
}
