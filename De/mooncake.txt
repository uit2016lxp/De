#include<stdio.h>
int main()
{
	int i,j,n,m;
	float a[3][100],t,w=0,e=0,p=0,u=0;
	scanf("%d %d",&n,&m);//月饼种类和市场需求
	for(i=0;i<2;i++)
		for(j=0;j<n;j++)
			scanf("%f",&a[i][j]);
	for(i=0;i<n;i++)
		a[2][i]=a[1][i]/a[0][i];//月饼单价
	for(j=0;j<n-1;j++)
		for(i=0;i<n-1-j;i++)
			if(a[2][i]<a[2][i+1])
			{
				t=a[2][i];
				a[2][i]=a[2][i+1];
				a[2][i+1]=t;
                                t=a[1][i];
				a[1][i]=a[1][i+1];
				a[1][i+1]=t;
				t=a[0][i];
				a[0][i]=a[0][i+1];
				a[0][i+1]=t;
			}//根据月饼的单价进行重新排序
	i=0;
	while(u<m)
	{
		w+=a[1][i];
		u+=a[0][i];
		i++;
	}
	w=w-(u-m)*a[2][i-1];
	printf("%.2f",w);
	return 0;
}
